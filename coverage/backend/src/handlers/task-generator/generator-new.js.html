function parseJsonArray(value) {
  if (!value) return null;
  try {
    const parsed = typeof value === "string" ? JSON.parse(value) : value;
    return Array.isArray(parsed) ? parsed : null;
  } catch (err) {
    console.warn("[TaskGenerator]  JSON ", value, err);
    return null;
  }
}

function isConfigEffectiveInMonth(config, targetYear, targetMonth) {
  if (!config.effective_month) return true;
  const [effYear, effMonth] = config.effective_month.split("-").map((n) => parseInt(n, 10));
  if (Number.isNaN(effYear) || Number.isNaN(effMonth)) return true;
  if (targetYear < effYear) return false;
  if (targetYear === effYear && targetMonth < effMonth) return false;
  return true;
}
 
/**
 * 
 */
function shouldGenerateInMonth(config, month, serviceContext) {
  const frequency = config.execution_frequency || config.delivery_frequency;
  const serviceMonths = serviceContext ? parseJsonArray(serviceContext.execution_months) : null;
  const executionMonths = parseJsonArray(config.execution_months);
 
  // 
  if (serviceMonths && Array.isArray(serviceMonths) && serviceMonths.length > 0) {
    return serviceMonths.includes(month);
  }
  if (executionMonths && Array.isArray(executionMonths)) {
    return executionMonths.includes(month);
  }
 
  // 
  switch (frequency) {
    case "monthly":
      return true; // 
 
    case "bi-monthly":
      return month % 2 === 1; // 1,3,5,7,9,11
 
    case "quarterly":
      return [1, 4, 7, 10].includes(month); // 
 
    case "semi-annual":
      return [1, 7].includes(month); // 1,7
 
    case "annual":
    case "yearly":
      return month === 1; // 1
 
    case "one-time":
      return false; // 
 
    default:
      return true; // 
  }
}

          primaryConfig.estimated_hours,  // 12

    } catch (err) {

'' : ''} ${generatedCount}  ${skippedCount}  ${actualProcessedServices}/${totalServices} ${queryCount}  D1  ${(durationMs / 1000).toFixed(2)}  ${(totalQueryTime / 1000).toFixed(2)}  ${(processingTime / 1000).toFixed(2)} ${!isComplete ? ` ${remainingServices} ` : ''}`
  );
  
  // 
  if (generatedCount > 0) {
    console.log(
      `[TaskGenerator]  ${timePerTask.toFixed(0)}ms ${queryTimePerTask.toFixed(0)}ms ${processingTimePerTask.toFixed(0)}ms ${generatedCount}  ${(durationMs / 1000).toFixed(2)} `
    );
  }
  
  //  breakdown
  if (queryTimings.length > 0) {
    console.log(`[TaskGenerator] ${JSON.stringify(queryTimings.map(t => `${t.query}:${t.duration}ms`))}`);
  }
 
  return { 
    generatedCount, 
    skippedCount,
    errors: errors.length,
    durationMs,
    processedServices: actualProcessedServices,
    totalServices: totalServices,
    remainingServices: remainingServices,
    isComplete: isComplete,
    queryCount: queryCount
  };
